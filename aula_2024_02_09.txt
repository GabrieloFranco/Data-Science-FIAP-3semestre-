
SELECT * FROM DB_CIDADE, DB_ESTADO

SELECT NM_ESTADO, NM_CIDADE
FROM DB_CIDADE
JOIN DB_ESTADO ON DB_CIDADE.SG_ESTADO = DB_ESTADO.SG_ESTADO
ORDER BY NM_ESTADO, NM_CIDADE ASC


SELECT CD_PRODUTO, DS_PRODUTO, VL_UNITARIO FROM DB_PRODUTO 

-----------------------------------

SELECT * FROM DB_PRODUTO,DB_CATEGORIA_PROD,DB_SUB_CATEGORIA_PROD

SELECT
DB_CATEGORIA_PROD.DS_CATEGORIA_PROD,
DB_SUB_CATEGORIA_PROD.DS_SUB_CATEGORIA_PROD,
DB_PRODUTO.CD_PRODUTO,
DB_PRODUTO.DS_PRODUTO,
DB_PRODUTO.VL_UNITARIO
FROM
    DB_CATEGORIA_PROD 
JOIN DB_SUB_CATEGORIA_PROD ON DB_CATEGORIA_PROD.CD_CATEGORIA_PROD = DB_SUB_CATEGORIA_PROD.CD_CATEGORIA_PROD
JOIN DB_PRODUTO ON DB_SUB_CATEGORIA_PROD.CD_SUB_CATEGORIA_PROD = DB_PRODUTO.CD_SUB_CATEGORIA_PROD
WHERE DS_SUB_CATEGORIA_PROD = 'Doce'
ORDER BY
 CD_PRODUTO,
 DS_CATEGORIA_PROD,
 DS_SUB_CATEGORIA_PROD
